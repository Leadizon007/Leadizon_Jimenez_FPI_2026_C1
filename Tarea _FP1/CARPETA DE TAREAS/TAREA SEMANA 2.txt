
#include <stdio.h>
#include <math.h>

int main(void) {
	double RES;
	int R, T, Q;

	printf("Ingrese los valores de R, T  y Q: ");
	scanf("%d %d %d", &R, &T, &Q);

	RES = pow(R, 4) - pow(T, 3) + 4 * pow(Q, 2);
	printf("\nRES = %.2f\n", RES);
	if (RES < 840) {
		printf("\nR = %d\tT = %d\tQ = %d\n", R, T, Q);
	}
	else {
		printf("No cumple la condiciC3n (RES >= 840)\n");
	}

	return 0;
}
