    int NUM;

    printf("Ingrese un número entero: ");

/*
Al recibir como datos tres valores enteros T, P y N,
el programa comprueba la igualdad de la expresiC3n:
T, P y N: variables de tipo entero.
*/

int main(void) {
	int T, P, N;


	printf("Ingrese los valores de T, P y N: ");
	scanf("%d %d %d", &T, &P, &N);

	if (P != 0) {
		double izquierda = pow((double)T / P, N);
		double derecha   = pow(T, N) / pow(P, N);

		if (fabs(izquierda - derecha) < 1e-6) {
			printf("\nSe comprueba la igualdad\n");
		} else {
			printf("\nNo se comprueba la igualdad\n");
		}
	} else {
		printf("\nP tiene que ser diferente de cero\n");
	}

	return 0;
}

/*
Programa en C que, al recibir como dato un valor entero Y,
calcula el resultado de la función:

F(X) =
    4/Y - Y      si 0 <= Y <= 10
    Y^3 - 12     si 11 < Y <= 25
    Y^2 + (Y^3 - 18) si 25 < Y <= 50
    0             para otro valor de Y

Y: variable de tipo entero
X: variable de tipo real
*/


int main(void) {
    float X;
    int Y;

    printf("Ingrese el valor de Y: ");
    scanf("%d", &Y);

 
    if (Y < 0 || Y > 50) {
        X = 0;
    } else if (Y <= 10) {
        if (Y != 0) { 
            X = 4.0 / Y - Y;
        } else {
            printf("Error: división\n");
            return 1;
        }
    } else if (Y <= 25) {
        X = pow(Y, 3) - 12;
    } else { // 25 < Y <= 50
        X = pow(Y, 2) + (pow(Y, 3) - 18);
    }

    printf("\nY = %d\tX = %8.2f\n", Y, X);

    return 0;
}

/*
Programa en C que calcula el costo de internación de un cliente
según el tipo de tratamiento, edad y días de estancia.

Tabla de costos por día:
1 -> 2800
2 -> 1950
3 -> 2500
4 -> 1150

Descuentos:
- Mayores de 60 años: 25%
- Menores de 25 años: 15%
*/

#include <stdio.h>

int main(void) {
    int TRA, EDA, DIA;
    float COS;

    printf("Ingrese tipo de tratamiento, edad y días: ");
    scanf("%d %d %d", &TRA, &EDA, &DIA);

    switch (TRA) {
        case 1: COS = DIA * 2800; break;
        case 2: COS = DIA * 1950; break;
        case 3: COS = DIA * 2500; break;
        case 4: COS = DIA * 1150; break;
        default: COS = -1; break;
    }

    if (COS != -1) {
        if (EDA >= 60) {
            COS = COS * 0.75; 
        } else if (EDA <= 25) {
            COS = COS * 0.85; 
        }

        printf("\nClave tratamiento: %d\t Días: %d\t Costo total: %8.2f\n",
               TRA, DIA, COS);
    } else {
        printf("\nLa clave del tratamiento es incorrecta\n");
    }

    return 0;
}
